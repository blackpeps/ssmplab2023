DATA SEGMENT
NUM1 DB 5EH
NUM2 DB 0EBH
DATA ENDS
CODE SEGMENT
ASSUME CS:CODE, DS:DATA
START: MOV AX, DATA
MOV DS, AX
MOV BH, NUM1
AND BH, NUM2 ; result is 4AH in BH
MOV BL, NUM1
OR BL, NUM2 ; result is FFH in BL
MOV CH, NUM1
XOR CH, NUM2 ; result is B5H in CH
MOV CL, NUM1
NOT CL ; result is A1H in CL
INT 3H ; DEBUGGING
CODE ENDS
END START
